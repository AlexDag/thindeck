47  /**
48   * Tests for {@link Lifecycle}.
49   *
50   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
51   * @version $Id$
52   */
53  public final class LifecycleTest {
54  
55      /**
56       * Lifecycle will add base attribute with correct class.
57       * @todo #319 When jcabi-aspects#95 is fixed enable this test.
58       */
59      @Test
60      @Ignore
61      public void initializedBaseAttribute() {
62          final ServletContextEvent event = Mockito
63              .mock(ServletContextEvent.class);
64          final ServletContext context = Mockito.mock(ServletContext.class);
65          Mockito.when(event.getServletContext()).thenReturn(context);
66          final Lifecycle lifecycle = new Lifecycle();
67          lifecycle.contextInitialized(event);
68          try {
69              Mockito.verify(context).setAttribute(
70                  Mockito.eq(Base.class.getName()),
71                  Mockito.argThat(Matchers.instanceOf(Base.class))
72              );
73          } finally {
74              lifecycle.contextDestroyed(event);
75          }
76      }
77  
78      /**
79       * RoutineTxn can increment transactions.
80       * @throws IOException In case of error.
81       * @todo #351 Implement MkScenario and MkStep.
82       *  Rewrite the test after that using mocks from
83       *  com.thindeck.api.mock package instead of mockito.
84       */
85      @Test
86      public void incrementsTransactionCount() throws IOException {
87          final Base base = Mockito.mock(Base.class);
88          final Repos repos = Mockito.mock(Repos.class);
89          final Repo repo = Mockito.mock(Repo.class);
90          final Tasks tasks = Mockito.mock(Tasks.class);
91          final Task task = Mockito.mock(Task.class);
92          final Txn txn = Mockito.mock(Txn.class);
93          Mockito.when(base.repos()).thenReturn(repos);
94          Mockito.when(repos.iterate()).thenReturn(Collections.singleton(repo));
95          Mockito.when(repo.tasks()).thenReturn(tasks);
96          Mockito.when(tasks.open()).thenReturn(Collections.singleton(task));
97          Mockito.when(base.txn(Mockito.any(Task.class))).thenReturn(txn);
98          final RoutineTxns routine = new RoutineTxns(base);
99          routine.close();
100         routine.run();
101         Mockito.verify(txn).increment();
102     }
103 }
